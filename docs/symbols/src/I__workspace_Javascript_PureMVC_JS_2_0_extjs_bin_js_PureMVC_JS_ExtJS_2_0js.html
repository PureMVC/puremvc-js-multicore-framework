337   /**
338    * @class <P>A base <code>Notification</code> implementation.</P>
339    * <P>
340    * PureMVC does not rely upon underlying event models such
341    * as the one provided with Flash, and ActionScript 3 does
342    * not have an inherent event model.</P>
343    *
344    * <P>
345    * The Observer pattern as implemented within PureMVC exists
346    * to support event-driven communication between the
347    * application and the actors of the MVC triad (Model, View
348    * and Controller.</P>
349    *
350    * <P>
351    * Notifications are not meant to be a replacement for Events
352    * in Flex/Flash/AIR/Javascript. Generally, <code>Mediator</code>
353    * implementors place event listeners on their view components, which they
354    * then handle in the usual way. This may lead to the broadcast of <code>Notification</code>s to
355    * trigger <code>Command</code>s or to communicate with other <code>Mediators</code>.
356    * <code>IProxy</code> and <code>Command</code>
357    * instances communicate with each other and <code>Mediator</code>s
358    * by broadcasting <code>Notification</code>s.</P>
359    *
360    * <P>
361    * A key difference between Flash <code>Event</code>s and PureMVC
362    * <code>Notification</code>s is that <code>Event</code>s follow the
363    * 'Chain of Responsibility' pattern, 'bubbling' up the display hierarchy
364    * until some parent component handles the <code>Event</code>, while
365    * PureMVC <code>Notification</code>s follow a 'Publish/Subscribe'
366    * pattern. PureMVC classes need not be related to each other in a
367    * parent/child relationship in order to communicate with one another
368    * using <code>Notification</code>s.
369    *
370    * @param {String} name the name of the notification.
371    * @param {Object} [body] (optional) body data to send with the notification.
372    * @param {String} [type] (optional) type identifier of the notification.
373    *
374    * @see Puremvc.patterns.Observer
375    * 
376    * @author Justin Wilaby
377    * @author Tony DeFusco
378    *
379    * @constructs
380    */
381   constructor: function(name /* String */, body /* Object */, type /* String */) {
382     Puremvc.patterns.Notification.superclass.constructor.call(this);
383     this.name = name;
384     this.body = body;
385     this.type = type;
386   },
387 
388   /**
389    * Get the name of the <code>Notification</code> instance.
390    *
391    * @return the name of the <code>Notification</code> instance.
392    * @type String
393    */
394   getName: function() {
395     return this.name;
396   },
397 
398   /**
399    * Set the body of the <code>Notification</code> instance.
400    *
401    * @param {Object} body the body of the notification.
402    */
403   setBody: function(body /* Object */) {
404     this.body = body;
405   },
406 
407   /**
408    * Get the body of the <code>Notification</code> instance.
409    *
410    * @return the body object.
411    * @type Object
412    */
413   getBody: function() {
414     return this.body;
415   },
416 
417   /**
418    * Set the type of the <code>Notification</code> instance.
419    *
420    * @param {String} type the type identifier for the notification.
421    */
422   setType: function(type /* String */) {
423     this.type = type;
424   },
425 
426   /**
427    * Get the type of the <code>Notification</code> instance.
428    *
429    * @return the type identifier for the notification.
430    * @type String
431    */
432   getType: function() {
433     return this.type;
434   },
435 
436   /**
437    * Get a textual representation of the <code>Notification</code> instance.
438    *
439    * @return the textual representation of the <code>Notification</code> instance.
440    * @type String
441    */
442   toString: function() {
443     var msg = "Notification Name: " + this.getName();
444     msg += "\nBody:" + ((this.body == null) ? "null": this.body.toString());
445     msg += "\nType:" + ((this.type == null) ? "null": this.type);
446     return msg;
447   }

510   /**
511    * @class <p>A base <code>Observer</code> implementation.</P>
512    * <P>
513    * An <code>Observer</code> is an object that encapsulates information
514    * about an interested object with a method that should
515    * be called when a particular <code>Notification</code> is broadcast.</P>
516    *
517    * <P>
518    * In PureMVC, the <code>Observer</code> class assumes these responsibilities:
519    * <UL>
520    * <LI>Encapsulate the notification (callback) method of the interested object.</LI>
521    * <LI>Encapsulate the notification context (this) of the interested object.</LI>
522    * <LI>Provide methods for setting the notification method and context.</LI>
523    * <LI>Provide a method for notifying the interested object.</LI>
524    * </UL>
525    *
526    * @see Puremvc.core.View
527    * @see Puremvc.patterns.Notification
528    *
529    * @param {Function} notifyMethod the notification method of the interested object.
530    * @param {Object} notifyContext the notification context of the interested object.
531    *
532    * @author Justin Wilaby
533    * @author Tony DeFusco
534    *
535    * @constructs
536    */
537   constructor: function(notifyMethod /* Function */, notifyContext /* Object */) {
538     Puremvc.patterns.Observer.superclass.constructor.call(this);
539     this.notify = notifyMethod;
540     this.context = notifyContext;
541   },
542 
543   /**
544    * Set the notification method.
545    *
546    * <P>
547    * The notification method should take one parameter of type <code>Notification</code>.</P>
548    *
549    * @param {Function} notifyMethod the notification (callback) method of the interested object.
550    */
551   setNotifyMethod: function(notifyMethod /* Function */) {
552     this.notify = notifyMethod;
553   },
554 
555   /**
556    * Set the notification context.
557    *
558    * @param {Object} notifyContext the notification context (this) of the interested object.
559    */
560   setNotifyContext: function(notifyContext /* Object */) {
561     this.context = notifyContext;
562   },
563 
564   /**
565    * Get the notification method.
566    *
567    * @return the notification (callback) method of the interested object.
568    * @type Function
569    */
570   getNotifyMethod: function() {
571     return this.notify;
572   },
573 
574   /**
575    * Get the notification context.
576    *
577    * @return the notification context (<code>this</code>) of the interested object.
578    * @type Object
579    */
580   getNotifyContext: function() {
581     return this.context;
582   },
583 
584   /**
585    * @private
586    * Notify the interested object.
587    *
588    * @param {Puremvc.patterns.Notification} notification the <code>Notification</code> to pass to the interested object's notification method.
589    */
590   notifyObserver: function(notification /* Notification */) {
591     this.notify.apply(this.context, [notification]);
592   },
593 
594   /**
595    * @private
596    * Compare an object to the notification context.
597    *
598    * @param {Object} object the object to compare.
599    *
600    * @return true if the object and the notification context are the same, otherwise false.
601    * @type Boolean
602    */
603   compareNotifyContext: function(object /* Object */) {
604     return object === this.context;
605   }
