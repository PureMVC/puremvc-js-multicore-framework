  8 /**
  9  * @misc
 10  * In PureMVC, the <code>Model</code> class provides
 11  * access to model objects (Proxies) by named lookup.
 12  * 
 13  * @class A Singleton <code>Model</code> implementation in the form
 14  * of a JSON object.
 15  * <P>
 16  * The <code>Model</code> assumes these responsibilities:</P>
 17  *
 18  * <UL>
 19  * <LI>Maintain a cache of <code>Proxy</code> instances.</LI>
 20  * <LI>Provide methods for registering, retrieving, and removing
 21  * <code>Proxy</code> instances.</LI>
 22  * </UL>
 23  *
 24  * <P>
 25  * Your application must register <code>Proxy</code> instances
 26  * with the <code>Model</code>. Typically, you use an
 27  * <code>SimpleCommand</code> to create and register <code>Proxy</code>
 28  * instances once the <code>Facade</code> has initialized the Core
 29  * actors.</p>
 30  *
 31  * @see Proxy
 32  * @author Justin Wilaby
 33  */
 34 var Model = function(){
 35 
 36     /**
 37      * @ignore
 38      * Constructor for MooTools Class creation
 39      */
 40     this.initialize = function()
 41     {
 42 	this.initializeModel();
 43     }
 44     /**
 45      * HashTable of <code>Proxy</code> instances
 46      * registered with the <code>Model</code>
 47      * @type Object
 48      */
 49     this.proxyMap = {}
 50 
 51     /**
 52      * Register an <code>Proxy</code> with the <code>Model</code>.
 53      *
 54      * @param {Proxy} proxy a <code>Proxy</code> to be held by the <code>Model</code>.
 55      */
 56     this.registerProxy = function(proxy /* Proxy */)
 57     {
 58 	this.proxyMap[proxy.getProxyName()] = proxy;
 59 	proxy.onRegister();
 60     }
 61 
 62     /**
 63      * Retrieve an <code>IProxy</code> from the <code>Model</code>.
 64      *
 65      * @param {String} proxyName The name of the <code>Proxy</code> to retrieve.
 66      * @returns {Proxy} The the <code>Proxy</code> instance previously registered with the given <code>proxyName</code>.
 67      */
 68     this.retrieveProxy = function(proxyName /* String */)
 69     {
 70 	return this.proxyMap[proxyName];
 71     }
 72 
 73     /**
 74      * Check if a Proxy is registered
 75      *
 76      * @param {String} proxyName
 77      * @returns {Boolean} whether a Proxy is currently registered with the given <code>proxyName</code>.
 78      */
 79     this.hasProxy = function(proxyName /* String */)
 80     {
 81 	return this.proxyMap[proxyName] != null;
 82     }
 83 
 84     /**
 85      * Remove an <code>Proxy</code> from the <code>Model</code>.
 86      *
 87      * @param {String} proxyName The name of the <code>Proxy</code> instance to be removed.
 88      * @returns {Proxy} the <code>Proxy</code> that was removed from the <code>Model</code>
 89      */
 90     this.removeProxy = function(proxyName /* String */)
 91     {
 92 	var proxy = this.proxyMap[proxyName];
 93 	if (proxy)
 94 	{
 95 	    delete this.proxyMap[proxyName];
 96 	    proxy.onRemove();
 97 	}
 98 	return proxy;
 99     }
100 
101     /**
102      * @ignore
103      * Omitted 
104      */
105     this.initializeModel = function()
106     {
107 
108     }
109 }

116     if (Model.instance == undefined)
117     {

120     }
121     return Model.instance;

130 /**
131 * @misc
132 * The View class in PureMVC.
133 * 
134 * @class A Singleton <code>View</code> implementation.
135 * In PureMVC, the <code>View</code> class assumes these responsibilities:
136 * <UL>
137 * <LI>Maintain a cache of <code>Mediator</code> instances.</LI>
138 * <LI>Provide methods for registering, retrieving, and removing <code>IMediators</code>.</LI>
139 * <LI>Notifiying <code>Mediators</code> when they are registered or removed.</LI>
140 * <LI>Managing the observer lists for each <code>Notification</code> in the application.</LI>
141 * <LI>Providing a method for attaching <code>Observers</code> to an <code>INotification</code>'s observer list.</LI>
142 * <LI>Providing a method for broadcasting an <code>Notification</code>.</LI>
143 * <LI>Notifying the <code>Observers</code> of a given <code>Notification</code> when it broadcast.</LI>
144 * </UL>
145 *
146 * @see Mediator
147 * @see Observer
148 * @see Notification
149 * @author Justin Wilaby
150 */
151 var View = function(){
152 
153     /**
154      * Mapping of <code>Mediator</code> names to
155      * <code>Mediator</code> instances
156      * @type Object
157      */
158     this.mediatorMap = {}
159     /**
160      * Mapping of <code>Notification</code> names to
161      * <code>Observers</code> lists
162      * @type Object
163      */
164     this.observerMap = {}
165 
166     /**
167     * Register an <code>Observer</code> to be notified
168     * of <code>Notifications</code> with a given name.
169     *
170     * @param {String} notificationName The name of the <code>Notifications</code> to notify this <code>Observer</code> of
171     * @param {Observer} observer The <code>Observer</code> to register.
172     */
173     this.registerObserver = function(notificationName /* String */, observer /* Observer */)
174     {
175 	var observers = this.observerMap[notificationName];
176 	if (observers)
177 	    observers.push(observer);
178 	else
179 	    this.observerMap[notificationName] = [observer];
180     }
181 
182     /**
183     * Notify the <code>Observers</code> for a particular <code>Notification</code>.
184     *
185     * <P>
186     * All previously attached <code>Observers</code> for this <code>Notification</code>'s
187     * list are notified and are passed a reference to the <code>Notification</code> in
188     * the order in which they were registered.</P>
189     *
190     * @param {Notification} notification The <code>Notification</code> to notify <code>Observers</code> of.
191     */
192     this.notifyObservers = function(notification /* Notification*/)
193     {
194 	var name = notification.getName();
195 	if (this.observerMap[name] == null)
196 	    return;
197 	// Copy the array
198 	var observers = this.observerMap[name].concat();
199 	var len = observers.length;
200 	for (var i = 0; i < len; i++)
201 	{
202 	    var observer = observers[i];
203 	    observer.notifyObserver(notification);
204 	}
205     }
206 
207     /**
208     * Remove the observer for a given notifyContext from an observer list for a given Notification name.
209     * <P>
210     * @param {String} notificationName which observer list to remove from
211     * @param {Object} notifyContext remove the observer with this object as its notifyContext
212     */
213     this.removeObserver = function(notificationName /* String */, notifyContext /* Object */)
214     {
215 	var observers = this.observerMap[notificationName];
216 	var i = observers.length;
217 	while(i--)
218 	{
219 	    var observer = observers[i];
220 	    if (observer.compareNotifyContext(notifyContext))
221 	    {
222 		observers.splice(i, 1);
223 		break;
224 	    }
225 	}
226 	// Remove empty observer lists.
227 	if (!observers.length)
228 	    delete this.observerMap[notificationName];
229     }
230 
231     /**
232     * Register an <code>IMediator</code> instance with the <code>View</code>.
233     *
234     * <P>
235     * Registers the <code>IMediator</code> so that it can be retrieved by name,
236     * and further interrogates the <code>IMediator</code> for its
237     * <code>INotification</code> interests.</P>
238     * <P>
239     * If the <code>IMediator</code> returns any <code>INotification</code>
240     * names to be notified about, an <code>Observer</code> is created encapsulating
241     * the <code>IMediator</code> instance's <code>handleNotification</code> method
242     * and registering it as an <code>Observer</code> for all <code>INotifications</code> the
243     * <code>IMediator</code> is interested in.</p>
244     *
245     * @param {Mediator} mediator a reference to the <code>Mediator</code> instance
246     */
247     this.registerMediator = function(mediator /* Mediator */)
248     {
249 	var name = mediator.getMediatorName();
250 	if (this.mediatorMap[name])
251 	    return;
252 	this.mediatorMap[name] = mediator;
253 	var interests = mediator.listNotificationInterests();
254 	var len = interests.length;
255 	if (len)
256 	{
257 	    var observer = new Observer(mediator.handleNotification, mediator);
258 	    for (var i = 0; i < len; i++)
259 		this.registerObserver(interests[i], observer);
260 	}
261 	mediator.onRegister();
262     }
263 
264     /**
265     * Retrieve a <code>Mediator</code> from the <code>View</code>.
266     *
267     * @param {String} mediatorName the name of the <code>IMediator</code> instance to retrieve.
268     * @return {Mediator} the <code>Mediator</code> instance previously registered with the given <code>mediatorName</code>.
269     */
270     this.retrieveMediator = function(mediatorName /* String */)
271     {
272 	return this.mediatorMap[mediatorName];
273     }
274 
275     /**
276     * Remove an <code>Mediator</code> from the <code>View</code>.
277     *
278     * @param {String} mediatorName name of the <code>IMediator</code> instance to be removed.
279     * @return {Mediator} the <code>Mediator</code> that was removed from the <code>View</code>
280     */
281     this.removeMediator = function(mediatorName /* String */)
282     {
283 	var mediator = this.mediatorMap[mediatorName];
284 	if (mediator)
285 	{
286 	    var interests = mediator.listNotificationInterests();
287 	    var i = interests.length;
288 	    while(i--)
289 		this.removeObserver(interests[i], mediator);
290 
291 	    delete this.mediatorMap[mediatorName];
292 	    mediator.onRemove();
293 	}
294 	return mediator;
295     }
296 
297     /**
298     * Check if a Mediator is registered or not
299     *
300     * @param {String} mediatorName
301     * @return {Boolean} whether a Mediator is registered with the given <code>mediatorName</code>.
302     */
303     this.hasMediator = function(mediatorName /* String */)
304     {
305 	return this.mediatorMap[mediatorName] != null;
306     }
307 }
308 /**
309  * Singleton implementation for the <code>View</code>
310  * @return {View} the Singleton instance of the <code>View</code>
311  */
312 View.getInstance = function()
313 {
314     if (View.instance == undefined)
315     {
316 	var classFactory = new Class(new View());
317 	View.instance = new classFactory();
318     }
319     return View.instance;
320 }

328 /**
329 * @misc
330 * The Controller class for PureMVC
331 * @class A Singleton <code>Controller</code> implementation.
332 * <P>
333 * In PureMVC, the <code>Controller</code> class follows the
334 * 'Command and Controller' strategy, and assumes these
335 * responsibilities:
336 * <UL>
337 * <LI> Remembering which <code>SimpleCommand</code>s
338 * or <code>MacroCommand</code>s
339 * are intended to handle which <code>Notifications</code>.</LI>
340 * <LI> Registering itself as an <code>Observer</code> with
341 * the <code>View</code> for each <code>Notification</code>
342 * that it has a <code>SimpleCommand</code>
343 * or <code>MacroCommand</code>  mapping for.</LI>
344 * <LI> Creating a new instance of the proper <code>SimpleCommand</code>
345 * or <code>MacroCommand</code>
346 * to handle a given <code>Notification</code> when notified by the <code>View</code>.</LI>
347 * <LI> Calling the <code>SimpleCommand</code>'s
348 * or <code>MacroCommand</code>'s  <code>execute</code>
349 * method, passing in the <code>Notification</code>.</LI>
350 * </UL>
351 *
352 * <P>
353 * Your application must register <code>ICommands</code> with the
354 * Controller.
355 * <P>
356 * The simplest way is to subclass </code>Facade</code>,
357 * and use its <code>initializeController</code> method to add your
358 * registrations.
359 *
360 * @see View
361 * @see Observer
362 * @see Notification
363 * @see SimpleCommand
364 * @see MacroCommand
365 * @author Justin Wilaby
366 */
367 var Controller = function(){
368 
369     /**
370      * The <code>View</code> singleton
371      * @type View
372      */
373     this.view = null;
374 
375     /**
376      * Mapping of <code>Notification<code> names to
377      * <code>Command</code> Mootools <code>Class</code> references
378      * @type Object
379      */
380     this.commandMap = {};
381 
382     /**
383      * @ignore
384      * MooTools Class constructor function.
385      */
386     this.initialize = function()
387     {
388 	 this.initializeController();
389     }
390 
391     /**
392      * Called automatically by the constructor.
393      * Retains a reference to the <code>View</code> singleton
394      */
395     this.initializeController = function()
396     {
397 	this.view = View.getInstance();
398     }
399     /**
400     * If an <code>SimpleCommand</code> or <code>MacroCommand</code>
401     * has previously been registered to handle a the given
402     * <code>Notification</code>, then it is executed.
403     *
404     * @param {Notification} note a <code>Notification</code>
405     */
406     this.executeCommand = function(note /* Notification */)
407     {
408 	var commandClassRef = this.commandMap[note.getName()];
409 	if (!commandClassRef)
410 	    return;
411 	var command = new commandClassRef();
412 	command.execute(note);
413     }
414 
415     /**
416     * Register a particular <code>SimpleCommand</code> or
417     * <code>MacroCommand</code> class as the handler
418     * for a particular <code>Notification</code>.
419     *
420     * <P>
421     * If an <code>Command</code> has already been registered to
422     * handle <code>Notification</code>s with this name, it is no longer
423     * used, the new <code>Command</code> is used instead.</P>
424     *
425     * The Observer for the new Command is only created if this the
426     * first time an Command has been regisered for this Notification name.
427     *
428     * @param notificationName the name of the <code>Notification</code>
429     * @param commandClassRef the <code>Class</code> of the <code>Command</code>
430     */
431     this.registerCommand = function(notificationName /* String */, commandClassRef /* Class */)
432     {
433 	if (!this.commandMap[notificationName])
434 	    this.view.registerObserver(notificationName, new Observer(this.executeCommand, this));
435 	this.commandMap[notificationName] = commandClassRef;
436     }
437 
438     /**
439      * Check if a Command is registered for a given Notification
440      *
441      * @param {String} notificationName
442      * @return {Boolean} whether a Command is currently registered for the given <code>notificationName</code>.
443      */
444     this.hasCommand = function(notificationName /* String */)
445     {
446 	return this.commandMap[notificationName] != null;
447     }
448 
449     /**
450      * Remove a previously registered <code>SimpleCommand</code>
451      * or <code>MacroCommand</code> to <code>Notification</code> mapping.
452      *
453      * @param {String} notificationName the name of the <code>Notification</code> to remove the
454      * <code>SimpleCommand</code> or <code>MacroCommand</code> mapping for
455      */
456     this.removeCommand = function(notificationName /* String */)
457     {
458 	if (!this.hasCommand(notificationName))
459 	    return;
460 	this.view.removeObserver(notificationName, this);
461 	delete this.commandMap[notificationName];
462     }
463 }
464 /**
465  * Singleton implementation for the <code>Controller</code>
466  * @return {Controller} the Singleton instance of the <code>Controller</code>
467  */
468 Controller.getInstance = function()
469 {
470     if (Controller.instance == undefined)
471     {
472 	var classFactory = new Class(new Controller());
473 	Controller.instance = new classFactory();
474     }
475     return Controller.instance;
476 }

484 /**
485  * @misc
486  * @class The Base <code>Notifier</code> implementation.
487  * <P>
488  * <code>MacroCommand, Command, Mediator</code> and <code>Proxy</code>
489  * all have a need to send <code>Notifications</code>. <P>
490  * <P>
491  * The <code>Notifier</code> base class provides a common method called
492  * <code>sendNotification</code> that relieves implementation code of
493  * the necessity to actually construct <code>Notifications</code>.</P>
494  *
495  * <P>
496  * The <code>Notifier</code> class, which all of the above mentioned classes
497  * extend, provides an initialized reference to the <code>Facade</code>
498  * Singleton, which is required for the convienience method
499  * for sending <code>Notifications</code>, but also eases implementation as these
500  * classes have frequent <code>Facade</code> interactions and usually require
501  * access to the facade anyway.</P>
502  *
503  * @see Facade
504  * @see Mediator
505  * @see Proxy
506  * @see SimpleCommand
507  * @see MacroCommand
508  * @author Justin Wilaby
509  */
510 var Notifier = function(){
511 
512     /**
513     * The <code>Facade</code> Singleton
514     * @type Facade
515     */
516     this.facade = null;
517 
518     /**
519      * @ignore
520      */
521     this.initialize = function()
522     {
523 	this.facade = Facade.getInstance();
524     }
525 
526     /**
527      * Create and send a <code>Notification</code>.
528      *
529      * <P>
530      * Keeps us from having to construct new Notification
531      * instances in our implementation code.
532      * @param {String} notificationName the name of the notiification to send
533      * @param {Object} [body] the body of the notification
534      * @param {String} [type] the type of the notification
535      */
536     this.sendNotification = function(notificationName /* String */, body /* Object */, type /* String */)
537     {
538 	this.facade.sendNotification(notificationName, body, type);
539     }
540 }

663 /**

665  * @class A base <code>Proxy</code> implementation.
666  * <P>
667  * In PureMVC, <code>Proxy</code> classes are used to manage parts of the
668  * application's data model. </P>

670  * <P>
671  * A <code>Proxy</code> might simply manage a reference to a local data object,

673  * getting of its data in synchronous fashion.</P>
674  *
675  * <P>
676  * <code>Proxy</code> classes are also used to encapsulate the application's
677  * interaction with remote services to save or retrieve data, in which case,
678  * we adopt an asyncronous idiom; setting data (or calling a method) on the
679  * <code>Proxy</code> and listening for a <code>Notification</code> to be sent
680  * when the <code>Proxy</code> has retrieved the data from the service. </P>
681  * @extends Notifier
682  * @param {String} proxyName The name of the Proxy
683  * @param {Object} data An initial data object to be held by the proxy.
684  * @see Model
685  * @author Justin Wilaby
686  */

699      * @type Object

702     
703     /**
704      * @ignore
705      */
706     this.initialize = function(proxyName /* String */, data /* Object */)

710 	this.data = data;
711     }
712

714      * Gets the proxyName

717      */

719     {
720 	return this.proxyName;
721     }
722 
723     /**
724      * Sets the data object
725      * @param {Object} data The data to set
726      */
727     this.setData = function(data /* Object */)
728     {

741     /**
742      * Called by the Model when the <code>Proxy</code> is registered.

1073 /**
1074  * @misc
1075  * @class A base <code>Notification</code> implementation.
1076  * <P>
1077  * PureMVC does not rely upon underlying event models such
1078  * as the one provided with Flash, and ActionScript 3 does
1079  * not have an inherent event model.</P>
1080  *
1081  * <P>
1082  * The Observer Pattern as implemented within PureMVC exists
1083  * to support event-driven communication between the
1084  * application and the actors of the MVC triad.</P>
1085  *
1086  * <P>
1087  * Notifications are not meant to be a replacement for Events
1088  * in Flex/Flash/Apollo. Generally, <code>Mediator</code> implementors
1089  * place event listeners on their view components, which they
1090  * then handle in the usual way. This may lead to the broadcast of <code>Notification</code>s to
1091  * trigger <code>Command</code>s or to communicate with other <code>Mediators</code>.
1092  * <code>IProxy</code> and <code>Command</code>
1093  * instances communicate with each other and <code>Mediator</code>s
1094  * by broadcasting <code>Notification</code>s.</P>
1095  *
1096  * <P>
1097  * A key difference between Flash <code>Event</code>s and PureMVC
1098  * <code>Notification</code>s is that <code>Event</code>s follow the
1099  * 'Chain of Responsibility' pattern, 'bubbling' up the display hierarchy
1100  * until some parent component handles the <code>Event</code>, while
1101  * PureMVC <code>Notification</code>s follow a 'Publish/Subscribe'
1102  * pattern. PureMVC classes need not be related to each other in a
1103  * parent/child relationship in order to communicate with one another
1104  * using <code>Notification</code>s.
1105  *

1107  * @param {Object} body data to send with the notification
1108  * @param {String} type type identifier of the notification

1213  * @misc
1214  * @class A base <code>Observer</code> implementation.
1215  * <P>
1216  * An <code>Observer</code> is an object that encapsulates information
1217  * about an interested object with a method that should
1218  * be called when a particular <code>Notification</code> is broadcast. </P>
1219  *
1220  * <P>
1221  * In PureMVC, the <code>Observer</code> class assumes these responsibilities:

1224  * <LI>Encapsulate the notification context (this) of the interested object.</LI>

1226  * <LI>Provide a method for notifying the interested object.</LI>
1227  * </UL>

1229  * @param {Object} notifyContext the notification context of the interested object
1230  * @see View
1231  * @see Notification

1329 /**
1330  * @misc
1331  * @class A base <code>Command</code> implementation that executes other <code>Command</code>s.
1332  * <P>
1333  * A <code>MacroCommand</code> maintains an list of
1334  * <code>ICommand</code> Class references called <i>SubCommands</i>.</P>
1335  *
1336  * <P>
1337  * When <code>execute</code> is called, the <code>MacroCommand</code>
1338  * instantiates and calls <code>execute</code> on each of its <i>SubCommands</i> turn.
1339  * Each <i>SubCommand</i> will be passed a reference to the original
1340  * <code>INotification</code> that was passed to the <code>MacroCommand</code>'s
1341  * <code>execute</code> method.</P>
1342  *
1343  * <P>
1344  * Unlike <code>SimpleCommand</code>, your subclass
1345  * should not override <code>execute</code>, but instead, should
1346  * override the <code>initializeMacroCommand</code> method,
1347  * calling <code>addSubCommand</code> once for each <i>SubCommand</i>
1348  * to be executed.</P>
1349  *
1350  * <P>
1351  * @extends Notifier
1352  * @see Controller
1353  * @see Notification
1354  * @see SimpleCommand
1355  */
1356 var MacroCommand = function(){
1357     this.Extends = Notifier;
1358 
1359     /**
1360      * An array of <code>SimpleCommands</code>
1361      * or subclasses of
1362      * @type Array
1363      */
1364     this.subCommands = [];
1365     /**
1366      * @ignore
1367      */
1368     this.initialize = function()
1369     {
1370 	this.initializeMacroCommand();
1371     }
1372 
1373     /**
1374      * Initialize the <code>MacroCommand</code>.
1375      *
1376      * <P>
1377      * In your subclass, override this method to
1378      * initialize the <code>MacroCommand</code>'s <i>SubCommand</i>
1379      * list with <code>ICommand</code> class references like
1380      * this:</P>
1381      *
1382      * <listing>
1383      *		// Initialize MyMacroCommand
1384      *		initializeMacroCommand : function()
1385      *		{
1386      *			this.addSubCommand(FirstCommand);
1387      *			this.addSubCommand(SecondCommand);
1388      *			this.addSubCommand(ThirdCommand);
1389      *		}
1390      * </listing>
1391      *
1392      * <P>
1393      * Note that <i>SubCommand</i>s may be any <code>Command</code> implementor,
1394      * <code>MacroCommand</code>s or <code>SimpleCommands</code> are both acceptable.
1395      */
1396     this.initializeMacroCommand = function(){}
1397 
1398     /**
1399      * Add a <i>SubCommand</i>.
1400      *
1401      * <P>
1402      * The <i>SubCommands</i> will be called in First In/First Out (FIFO)
1403      * order.</P>
1404      *
1405      * @param {Class} commandClassRef a reference to the <code>Class</code> of the <code>ICommand</code>.
1406      */
1407     this.addSubCommand = function(commandClassRef /* Class */)
1408     {
1409 	this.subCommands.push(commandClassRef);
1410     }
1411 
1412     /**
1413      * Execute this <code>MacroCommand</code>'s <i>SubCommands</i>.
1414      *
1415      * <P>
1416      * The <i>SubCommands</i> will be called in First In/First Out (FIFO)
1417      * order.
1418      *
1419      * @param {Notification} notification the <code>Notification</code> object to be passsed to each <i>SubCommand</i>.
1420      */
1421     this.execute = function(notification /* Notification */)
1422     {
1423 	var len = this.subCommands.length;
1424 	for (var i = 0; i < len; i++)
1425 	{
1426 	    var commandClassRef = this.subCommands[i];
1427 	    var commandInstance = new commandClassRef();
1428 	    commandInstance.execute(notification);
1429 	}
1430     }
1431 }
